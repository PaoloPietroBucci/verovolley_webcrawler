user_clips(user_id: str, amount: int = 50)
    - Get list of clips (reels) by user_id
    - List[Media]

usertag_medias(user_id: str, amount: int = 20)
    - Get medias where a user is tagged
    - List[Media]

media_likers(media_id: str)
    - Return list of users who liked this post (due to Instagram limitations, this may not return a complete list)
    - List[UserShort]

user_followers(user_id: str, amount: int = 0)
    - Get dict of followers users (amount=0 - fetch all followers)
    - Dict[int, UserShort]

user_following(user_id: str, amount: int = 0)
    - Get dict of following users (amount=0 - fetch all)
    - Dict[int, UserShort]

media_comments_chunk(media_id: str, max_amount: int, min_id: str = None)
    - Get chunk of comments on a media and end_cursor
    - Tuple[List[Comment], str]

user_stories(user_id: str, amount: int = None)
    - Get list of stories by user_id
    - List[Story]

story_info(story_pk: int, use_cache: bool = True)
    - Return story info
    - Story
